MACD DUPLO 12/72
Filtro direcional do sistema. EMA 12 (rápida/vermelha) vs EMA 72 (lenta/azul) + histograma. Períodos configuráveis. 
//@version=5
indicator("MACD Duplo 12/72 [ESPECULADOR]", overlay=false)

// === INPUTS (configuráveis) ===
len_fast = input.int(12, "EMA Rápida", minval=1)
len_slow = input.int(72, "EMA Lenta", minval=1)
src = input.source(close, "Fonte")
col_fast = input.color(color.red, "Cor Rápida")
col_slow = input.color(color.blue, "Cor Lenta")
col_hist_up = input.color(color.new(#00E676, 30), "Histograma +")
col_hist_dn = input.color(color.new(#FF5252, 30), "Histograma -")

// === CÁLCULO ===
ema_fast = ta.ema(src, len_fast)
ema_slow = ta.ema(src, len_slow)
diff = ema_fast - ema_slow

// === PLOTS ===
plot(ema_fast, "EMA Rápida", color=col_fast, linewidth=2)
plot(ema_slow, "EMA Lenta", color=col_slow, linewidth=2)
plot(diff, "Histograma", style=plot.style_histogram,
     color=diff >= 0 ? col_hist_up : col_hist_dn, linewidth=2)
hline(0, "Zero", color=color.gray, linestyle=hline.style_dashed)

// === ALERTAS ===
bull_cross = ta.crossover(ema_fast, ema_slow)
bear_cross = ta.crossunder(ema_fast, ema_slow)

plotshape(bull_cross, "Cruzamento Bull", shape.triangleup,
          location.bottom, color.new(#00E676, 0), size=size.small)
plotshape(bear_cross, "Cruzamento Bear", shape.triangledown,
          location.top, color.new(#FF5252, 0), size=size.small)

alertcondition(bull_cross, "MACD Bull Cross",
               "EMA rápida cruzou ACIMA da lenta — viés COMPRADOR")
alertcondition(bear_cross, "MACD Bear Cross",
               "EMA rápida cruzou ABAIXO da lenta — viés VENDEDOR")

RSI DUPLO
Dois RSIs sobrepostos para cruzamento de momentum. RSI rápido (curto prazo) vs RSI lento (longo prazo). Períodos configuráveis. 
//@version=5
indicator("RSI Duplo [ESPECULADOR]", overlay=false)

// === INPUTS (configuráveis) ===
len_fast = input.int(7, "RSI Rápido", minval=1)
len_slow = input.int(14, "RSI Lento", minval=1)
src = input.source(close, "Fonte")
col_fast = input.color(color.red, "Cor Rápido")
col_slow = input.color(color.blue, "Cor Lento")
ob_level = input.int(70, "Sobrecompra", minval=50, maxval=100)
os_level = input.int(30, "Sobrevenda", minval=0, maxval=50)

// === CÁLCULO ===
rsi_fast = ta.rsi(src, len_fast)
rsi_slow = ta.rsi(src, len_slow)

// === PLOTS ===
plot(rsi_fast, "RSI Rápido", color=col_fast, linewidth=2)
plot(rsi_slow, "RSI Lento", color=col_slow, linewidth=2)

h_ob = hline(ob_level, "Sobrecompra", color=color.new(#FF5252, 60),
             linestyle=hline.style_dashed)
h_os = hline(os_level, "Sobrevenda", color=color.new(#00E676, 60),
             linestyle=hline.style_dashed)
hline(50, "Meio", color=color.new(color.gray, 70),
      linestyle=hline.style_dotted)

fill(h_ob, h_os, color=color.new(color.gray, 92))

// === CRUZAMENTOS ===
bull_cross = ta.crossover(rsi_fast, rsi_slow)
bear_cross = ta.crossunder(rsi_fast, rsi_slow)

plotshape(bull_cross and rsi_fast < 50, "RSI Bull Cross",
          shape.circle, location.bottom, color.new(#00E676, 0),
          size=size.tiny)
plotshape(bear_cross and rsi_fast > 50, "RSI Bear Cross",
          shape.circle, location.top, color.new(#FF5252, 0),
          size=size.tiny)

// === BACKGROUND ===
bgcolor(rsi_fast > ob_level and rsi_slow > ob_level ?
        color.new(#FF5252, 92) : na)
bgcolor(rsi_fast < os_level and rsi_slow < os_level ?
        color.new(#00E676, 92) : na)

// === ALERTAS ===
alertcondition(bull_cross, "RSI Bull Cross",
               "RSI rápido cruzou ACIMA do lento — momentum comprador")
alertcondition(bear_cross, "RSI Bear Cross",
               "RSI rápido cruzou ABAIXO do lento — momentum vendedor")
alertcondition(rsi_fast > ob_level and rsi_slow > ob_level,
               "Dupla Sobrecompra", "Ambos RSIs em sobrecompra")
alertcondition(rsi_fast < os_level and rsi_slow < os_level,
               "Dupla Sobrevenda", "Ambos RSIs em sobrevenda")

DEMA 305 / 610
Médias DEMA como suporte/resistência dinâmico. Preço acima = comprador, abaixo = vendedor. Overlay no gráfico. 
//@version=5
indicator("DEMA 305/610 [ESPECULADOR]", overlay=true)

// === INPUTS (configuráveis) ===
len1 = input.int(305, "DEMA Curta", minval=1)
len2 = input.int(610, "DEMA Longa", minval=1)
src = input.source(close, "Fonte")
col1 = input.color(color.new(#00BCD4, 0), "Cor DEMA Curta")
col2 = input.color(color.new(#FFD740, 0), "Cor DEMA Longa")

// === CÁLCULO DEMA ===
dema(source, length) =>
    e1 = ta.ema(source, length)
    e2 = ta.ema(e1, length)
    2 * e1 - e2

dema_short = dema(src, len1)
dema_long = dema(src, len2)

// === PLOTS ===
p1 = plot(dema_short, "DEMA 305", color=col1, linewidth=2)
p2 = plot(dema_long, "DEMA 610", color=col2, linewidth=2)
fill(p1, p2, color=dema_short > dema_long ?
     color.new(#00E676, 90) : color.new(#FF5252, 90))

// === ALERTAS ===
alertcondition(ta.crossover(dema_short, dema_long),
               "DEMA Bull Cross", "DEMA 305 cruzou ACIMA da 610")
alertcondition(ta.crossunder(dema_short, dema_long),
               "DEMA Bear Cross", "DEMA 305 cruzou ABAIXO da 610")
